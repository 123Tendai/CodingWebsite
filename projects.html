import random


# Ask user their name to start with the game
# Use random numbers
# Having only a limit of 3 guess counts
name = input("Enter your name: ")
print("Get started", name, "..........")
print("\n"*2)
lucky_Number = random.randint(1, 50)
print("________________________________________________________________________________")
print("Enter a random number in range of 1-50 to WIN!!!!")
print("________________________________________________________________________________")

for guess_count in range(1, 4):
print("Guess a random number")
guess = int(input())
if guess < lucky_Number:
print("Guessed number is too low")
elif guess > lucky_Number:
print("Guessed number is too high")
else:
break
if guess == lucky_Number:
print("WELL DONE, you guessed the random number in", guess_count, " guesses")
else:
print("Failed, Try Again")
print("************************************")
print(lucky_Number, " was the random number")
